#include <iostream>
#include <cmath>
#include <string>
using namespace std;

enum enButtons {Withdraw = 1,Deposite, Exit};

int ReadPositiveNumber(string msg)
{
	int number;
	do
	{
		cout << msg;
		cin >> number;
		cout << endl;
	} while (number <= 0);
	
	return number;
}
int ReadPIN(string msg)
{
	int PIN;
	cout << msg;
	cin >> PIN;

	return PIN;
}
bool CheckPIN(int pin)
{
	return (pin == 1234);
}
string UserAccess(int pin)
{
	switch (CheckPIN(pin))
	{
	case true:
		return "Balance is 7500";
	case false:
		return "Wrong PIN";
	}
}

int ReadButtonID(string msg)
{
	int button_id;
	cout << msg << endl;
	cin >> button_id;

	return button_id;
}
enButtons Buttons()
{
	int button_id;
	button_id = ReadButtonID("(1) to Withdraw  (2) to Deposite  (3) Exit ");

	return (enButtons)button_id;
}
string UserOptions(enButtons select)
{
	switch (select)
	{
	case enButtons::Withdraw:
		return "How much you want? ";
	case enButtons::Deposite:
		return "How much to deposite? ";
	case enButtons::Exit:
		return "Bye!";
	default:
		return "Invalid input";
	}

}

bool IsValid(int balance, int requested_amount)
{
	if (requested_amount > balance || requested_amount <= 0)
		return false;
	else
		return true;
}
string ProcessWithdrawal(int current_balance, int requested_amount)
{
	if (IsValid(current_balance, requested_amount))
	{
		int remaining_balance = current_balance - requested_amount;
		return "Success! Remaining balance: " + to_string(remaining_balance) + " $";
	}
	else {
		return "Insufficient balance or invalid amount!";
	}

}

string ProcessDeposite(int current_balance, int added_amount)
{
	current_balance = current_balance + added_amount;
	return "Success! your new balance is: " + to_string(current_balance) + " $";
}
void Transaction(enButtons button_id)
{
	if (button_id == enButtons::Withdraw)
	{
		int requested_amount = ReadPositiveNumber("Enter amount: ");
		string process_withdrawal = ProcessWithdrawal(7500, requested_amount);
		cout << process_withdrawal << endl;
	}
	else if (button_id == enButtons::Deposite)
	{
		int deposite_amount = ReadPositiveNumber("Enter your bills: ");
		string process_deposite = ProcessDeposite(7500, deposite_amount);
		cout << process_deposite << endl;
	}
	
}









int main()
{
	int pin = ReadPIN("Please Enter you PIN? ");
	cout << UserAccess(pin) << endl;

	if (pin != 1234)
		return 0;

	enButtons button_id = Buttons();
	string user_options = UserOptions(button_id);
	cout << user_options << endl;
	
	Transaction(button_id);
	


	

	return 0;
}
