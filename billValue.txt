#include <iostream>
#include <cmath>
#include <string>
using namespace std;

float ReadPositiveNumber(string msg)
{
	float number;
	do
	{
		cout << msg;
		cin >> number;
		cout << endl;
	} while (number < 0);
	
	return number;
}
float ServiceFee(float bill_value)
{
	return bill_value * 0.10;
}
float SubTotal(float bill_value, float services)
{
	return bill_value + services;
}
float TaxValue(float sub_total)
{
	return sub_total * 0.16;
}
float TotalBill(float sub_total, float tax_value)
{
	return sub_total + tax_value;
}
float DiscountValue(float total, float discount)
{
	return total * (discount / 100);
}
void PrintBill()
{
	float bill_value = ReadPositiveNumber("Enter bill value: ");
	float discount = ReadPositiveNumber("Discount (%): ");
	float Services = ServiceFee(bill_value);
	float sub_total = SubTotal(bill_value, Services);
	float tax_value = TaxValue(sub_total);
	float Total_bill = TotalBill(sub_total, tax_value);
	float discount_value = DiscountValue(Total_bill, discount);
	float bill_final = Total_bill - discount_value;
	
	cout << "Sub Total         = " << bill_value << "$" << endl;
	cout << "Service Fee (10%) = " << Services << "$" << endl;
	cout << "Tax value (16%)   = " << tax_value << "$" << endl;
	cout << "Discount (" << discount << ")      = " << discount_value << "$" << endl;
	cout << "----------------------------" << endl;
	cout << "Total             = " << bill_final << "$" << endl;

}



int main()
{
	
	PrintBill();
	




	return 0;
}
