#include <iostream>
#include <cmath>
#include <string>
using namespace std;

struct Data
{
	int days, hours, minutes, secs;
};

int ReadPositiveNumber(string msg)
{
	int number;
	do
	{
		cout << msg;
		cin >> number;
		cout << endl;
	} while (number < 0);
	
	return number;
}
Data FillData(int seconds)
{
	
	Data fill;
	const int SecondsPerDay = 24 * 60 * 60;
	const int SecondsPerHour = 60 * 60;
	const int SecondsPerMinute = 60;

	int remainder;
	fill.days = seconds / SecondsPerDay;
	remainder = seconds % SecondsPerDay;
	fill.hours = remainder / SecondsPerHour;
	remainder = remainder % SecondsPerHour;
	fill.minutes = remainder / SecondsPerMinute;
	remainder = remainder % SecondsPerMinute;
	fill.secs = remainder;

	return fill;
}
void PrintResult(Data res)
{
	
	cout << res.days << ":"
		<< res.hours << ":"
		<< res.minutes << ":"
		<< res.secs << endl;

}





int main()
{
	int total_seconds = ReadPositiveNumber("Enter total seconds? ");
	Data result = FillData(total_seconds);

	PrintResult(result);
	



	
}
