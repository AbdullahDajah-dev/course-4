#include <iostream>
#include <cmath>
#include <string>
using namespace std;

enum enPrime { Prime = 1, NotPrime };
float CheckPositive(string msg)
{
	int number;
	do
	{
		cout << msg;
		cin >> number;
	} while (number <= 0);

	return number;
}
enPrime CheckPrime(int number)
{
	int M = round(number / 2);
	for (int counter = 2; counter <= M;counter++)
		if (number % counter == 0)
		{
			return enPrime::NotPrime;
		}
	return enPrime::Prime;
}
void PrintNumberType(int number)
{
	switch (CheckPrime(number))
	{
	case enPrime::Prime:
		cout << "\n(" << number << ") is a prime number." << endl;
		break;
	case enPrime::NotPrime:
		cout << "\n(" << number << ") is NOT a prime number." << endl;
		break;
	}
}





int main()
{
	
	
	float number = CheckPositive("Enter positive number: ");
	PrintNumberType(number);




	return 0;
}
